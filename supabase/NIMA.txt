-- WARNING: This schema is for context only and is not meant to be run.
-- Table order and constraints may not be valid for execution.

CREATE TABLE public.persona (
  idpersona uuid NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  nombres character varying NOT NULL,
  apellidos character varying NOT NULL,
  celular text CHECK (celular IS NULL OR length(celular) = 9),
  fecha_nacimiento date,
  tipo_documento USER-DEFINED NOT NULL,
  numero_documento text UNIQUE CHECK (length(numero_documento) <= 12),
  CONSTRAINT persona_pkey PRIMARY KEY (idpersona),
  CONSTRAINT persona_idpersona_fkey FOREIGN KEY (idpersona) REFERENCES auth.users(id)
);
CREATE TABLE public.paciente (
  idpaciente uuid NOT NULL,
  codigo_universitario character varying NOT NULL UNIQUE,
  tipo_usuario USER-DEFINED NOT NULL,
  facultad_escuela character varying NOT NULL,
  direccion character varying,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT paciente_pkey PRIMARY KEY (idpaciente),
  CONSTRAINT paciente_idpaciente_fkey FOREIGN KEY (idpaciente) REFERENCES public.persona(idpersona)
);
CREATE TABLE public.fisioterapeuta (
  idfisioterapeuta uuid NOT NULL DEFAULT auth.uid(),
  tipo_personal USER-DEFINED,
  especialidad character varying NOT NULL,
  numero_colegiatura character varying,
  activo boolean NOT NULL DEFAULT true,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT fisioterapeuta_pkey PRIMARY KEY (idfisioterapeuta),
  CONSTRAINT personal_salud_idpersona_fkey FOREIGN KEY (idfisioterapeuta) REFERENCES public.persona(idpersona)
);
CREATE TABLE public.enfermera (
  idenfermera uuid NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT enfermera_pkey PRIMARY KEY (idenfermera),
  CONSTRAINT secretaria_idsecretaria_fkey FOREIGN KEY (idenfermera) REFERENCES public.persona(idpersona)
);
CREATE TABLE public.servicio_topico (
  idservicio bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  nombre_servicio character varying NOT NULL,
  descripcion text,
  duracion_estimada_minutos integer DEFAULT 20,
  activo boolean NOT NULL DEFAULT true,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT servicio_topico_pkey PRIMARY KEY (idservicio)
);
CREATE TABLE public.horario (
  idhorario bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  idfisioterapeuta uuid NOT NULL,
  dia_semana smallint NOT NULL CHECK (dia_semana >= 1 AND dia_semana <= 6),
  hora_inicio time without time zone NOT NULL,
  hora_fin time without time zone NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT horario_pkey PRIMARY KEY (idhorario),
  CONSTRAINT horario_idpersonalsalud_fkey FOREIGN KEY (idfisioterapeuta) REFERENCES public.fisioterapeuta(idfisioterapeuta)
);
CREATE TABLE public.cita (
  idcita bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  idpaciente uuid NOT NULL,
  idfisioterapeuta uuid,
  idservicio bigint NOT NULL,
  fecha_hora timestamp with time zone NOT NULL,
  motivo_consulta text,
  estado USER-DEFINED NOT NULL DEFAULT 'pendiente'::estado_cita,
  paciente_en_sala boolean NOT NULL DEFAULT false,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT cita_pkey PRIMARY KEY (idcita),
  CONSTRAINT cita_idpaciente_fkey FOREIGN KEY (idpaciente) REFERENCES public.paciente(idpaciente),
  CONSTRAINT cita_idpersonalsalud_fkey FOREIGN KEY (idfisioterapeuta) REFERENCES public.fisioterapeuta(idfisioterapeuta),
  CONSTRAINT cita_idservicio_fkey FOREIGN KEY (idservicio) REFERENCES public.servicio_topico(idservicio)
);
CREATE TABLE public.Sesion (
  idatencion bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  idcita bigint NOT NULL UNIQUE,
  idpaciente uuid NOT NULL,
  idfisioterapeuta uuid NOT NULL,
  fecha_atencion timestamp with time zone NOT NULL DEFAULT now(),
  presion_arterial character varying,
  peso_kg numeric,
  talla_cm integer,
  temperatura_c numeric,
  frecuencia_cardiaca integer,
  saturacion_oxigeno integer,
  sintomas text NOT NULL,
  diagnostico_cie10 character varying,
  diagnostico_descripcion text NOT NULL,
  tratamiento_recetado text,
  observaciones_internas text,
  CONSTRAINT Sesion_pkey PRIMARY KEY (idatencion),
  CONSTRAINT atencion_clinica_idcita_fkey FOREIGN KEY (idcita) REFERENCES public.cita(idcita),
  CONSTRAINT atencion_clinica_idpaciente_fkey FOREIGN KEY (idpaciente) REFERENCES public.paciente(idpaciente),
  CONSTRAINT atencion_clinica_idpersonalsalud_fkey FOREIGN KEY (idfisioterapeuta) REFERENCES public.fisioterapeuta(idfisioterapeuta)
);
CREATE TABLE public.antecedentes (
  idantecedentes bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  idpaciente uuid NOT NULL UNIQUE,
  alergias text DEFAULT 'Ninguna'::text,
  antecedentes_familiares text,
  enfermedades_cronicas text,
  operaciones_quirurgicas text,
  grupo_sanguineo character varying,
  observaciones_generales text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT antecedentes_pkey PRIMARY KEY (idantecedentes),
  CONSTRAINT antecedentes_idpaciente_fkey FOREIGN KEY (idpaciente) REFERENCES public.paciente(idpaciente)
);
CREATE TABLE public.historial_estado_cita (
  idhistorial bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  idcita bigint NOT NULL,
  estado_anterior character varying NOT NULL,
  estado_nuevo character varying NOT NULL,
  fecha_cambio timestamp with time zone NOT NULL DEFAULT now(),
  usuario_responsable uuid NOT NULL,
  observacion text,
  CONSTRAINT historial_estado_cita_pkey PRIMARY KEY (idhistorial),
  CONSTRAINT historial_estado_cita_idcita_fkey FOREIGN KEY (idcita) REFERENCES public.cita(idcita)
);
